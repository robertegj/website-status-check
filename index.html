    <div id="siteChecker"></div>
    
    <script>
      // Site Checker Logic - edit or add items to this list
      var sitesWithImages = [
        "https://www.deliverlogic.com/favicon.ico",
        "https://www.deliverlogic.com/favicon.ico"
      ];
    
      function imageExists(image_url){
        var http = new XMLHttpRequest();
  
        http.open('HEAD', image_url, false);
        http.send();
  
        return http.status != 404;
      }
  
      siteCheckerDiv = document.getElementById("siteChecker");
  
      for(var i = 0; i < sitesWithImages.length; i++){
        item = document.createElement("ul");
        dot = document.createElement("span");
        dot.classList.add("dot");
        domain = new URL(sitesWithImages[i]);
        item.innerText = domain.hostname;
        item.appendChild(dot);
        if(imageExists(sitesWithImages[i])){
          dot.style.background = "green";
        }
        siteCheckerDiv.appendChild(item);
      }
    </script>
